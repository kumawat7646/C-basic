#include <stdio.h>

int main()
{
    int a,b,i;
    printf("enter no. to check prime or not :");
    scanf("%d",&a);
    
    for(i=1; i<=a; i++)
    {
	if(a%i==0)
	{
	b++;	
	}
    }
	
    if(b==2)
    {
	printf("it is a prime no.");
    }
    else
    {
	printf("it is not a prime no.");
    }

    return 0;
}

#include <stdio.h>

int main()
{
    int a,i,j;
    printf("enter the pattern value:");
    scanf("%d",&a);
    
    for(i=1; i<=a; i++)
    {
        for(j=1; j<=i; j++)
        {
            printf("*");
        }
    printf("\n");
    }
    

    return 0;
}
